def get_portfolio_ret_std( df ):
  mean_rets = np.asmatrix( np.mean( df , axis = 1 ) )

  weights = np.asmatrix( get_stock_weights(df.shape[0]) )
  stock_covr = np.asmatrix( np.cov( df ) )

  p_ret = weights * mean_rets.T
  p_sigma = np.sqrt( weights * stock_covr * weights.T )

  return weights.tolist()[0], p_ret.tolist()[0], p_sigma.tolist()[0]
